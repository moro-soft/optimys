${typeName}

${Object.keys(editingTsTypes[typeName]).map(key => ` `).join('')}
`; container.appendChild(typeDiv); }); } // Добавить новый тип ТС function addNewTsType() { const newTypeName = prompt('Введите название нового типа ТС:'); if (!newTypeName) return; if (editingTsTypes[newTypeName]) { alert('Тип с таким названием уже существует!'); return; } // Добавляем с параметрами по умолчанию editingTsTypes[newTypeName] = { L: 1200, T: 240, H: 260, maxW: 25000, maxCML: 30, maxCMT: 10, maxKGL: 1500, maxKGT: 500, minCMDoor: 50, minDimX: 25 }; renderTsTypesList(); } // Удалить тип ТС function deleteTsType(typeName) { if (!confirm(`Удалить тип "${typeName}"?`)) return; delete editingTsTypes[typeName]; renderTsTypesList(); } // Сохранить изменения типов ТС async function saveTsTypesChanges() { if (await updateTsIni(editingTsTypes)) { closeTsTypesModal(); } else { alert('Не удалось сохранить изменения'); } }
